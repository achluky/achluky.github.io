\begin{frame}{Why: Task Abstraction?}
    \begin{block}{Tujuan Pembelajaran}
        Mampu mengabstraksi tugas dari domain spesifik menjadi kosakata umum (\textit{Action} \& \textit{Target}) untuk menentukan solusi visual yang tepat.
    \end{block}

    \vspace{0.4cm}
    \textbf{Topik Pembahasan Utama:}
    \begin{enumerate}
        \item \textbf{High-level Actions:} Mengonsumsi vs Memproduksi data.
        \item \textbf{Search Levels:} Strategi mencari berdasarkan pengetahuan lokasi dan target.
        \item \textbf{Query Levels:} Tingkat kedalaman pertanyaan (Identifikasi $\rightarrow$ Komparasi $\rightarrow$ Ringkasan).
        \item \textbf{Targets:} Fokus analisis pada tren, outlier, atau fitur spesifik dataset.
    \end{enumerate}
\end{frame}

% ------------------------------
\begin{frame}{Why: Task Abstraction?}
    \vspace{-0.3cm}
    \begin{block}{Task Abstraction}
        	Tentang memahami niat di balik sebuah visualisasi. Jika pertemuan sebelumnya bertanya "Data apa yang kita punya?", maka saat ini bertanya "Apa yang ingin dicapai pengguna dengan data tersebut?".
    \end{block}
    \vspace{0.3cm}
	\begin{itemize}
		\item Why: Task Abstraction
		\begin{itemize}
			\item Actions (Tindakan)
				\begin{itemize}
					\item Analyze (Consume, Produce)
					\item Search (Lookup, Locate, Browse, Explore)
					\item Query (Identify, Compare, Summarize)
				\end{itemize}
			\item Targets (Sasaran)
				\begin{itemize}
					\item All Data (Trends, Outliers, Features)
					\item Attributes (Distribution, Extremes, Correlation)
				\end{itemize}
		\end{itemize}
	\end{itemize}	    
    
    \centering \textit{"Abstraksi tugas memungkinkan kita merancang visualisasi yang berfokus pada tujuan pengguna, bukan hanya pada ketersediaan data."}
\end{frame}

% ------------------------------
\begin{frame}{Mengapa Abstraksi Tugas Diperlukan?}
    \begin{block}{Tujuan Utama}
        Mentransformasi kebutuhan pengguna dari \textit{Domain-Specific} menjadi \textit{Abstract Tasks}.
    \end{block}

    \vspace{0.3cm}
    \textbf{Keuntungan Utama:}
    \begin{itemize}
        \item \textbf{Generalisasi:} Solusi untuk masalah keuangan dapat diterapkan pada masalah medis jika struktur tugasnya sama.
        \item \textbf{Kosakata Umum:} Memungkinkan desainer dan pengguna dari latar belakang berbeda untuk berkomunikasi.
        \item \textbf{Efisiensi Desain:} Mempersempit pilihan teknik visualisasi ke arah yang paling efektif secara kognitif.
        \item \textbf{Analisis Komparatif:} Memudahkan evaluasi apakah sebuah \textit{tool} benar-benar menyelesaikan masalah inti pengguna.
    \end{itemize}

    \vspace{0.3cm}
    \centering \textit{"Analyze the task, not just the data."}
\end{frame}

% ------------------------------
\begin{frame}{Mengapa Abstraksi Tugas Diperlukan?}
	\begin{enumerate}
		\item Menghindari "Reinventing the Wheel" (Mencegah Pengulangan) \\
		Terlihat berbeda, padahal secara struktural sama. \\ \textbf{Contoh:} Seorang analis bursa saham ingin mencari "lonjakan harga yang aneh", sementara seorang ahli epidemiologi ingin mencari "wilayah dengan lonjakan kasus penyakit yang tidak biasa". \\ \textbf{Abstraksi:} Keduanya memiliki tugas yang sama, yaitu Locate Outliers. Dengan abstraksi, kita bisa menggunakan solusi visual yang sudah terbukti efektif untuk outliers di satu bidang ke bidang lainnya tanpa memulai dari nol.
		\item Memungkinkan Perbandingan Lintas Domain \\
		Menyediakan kosakata umum (shared vocabulary). Kosakata umum ini, peneliti visualisasi dapat membandingkan sistem yang dirancang untuk ahli biologi dengan sistem yang dirancang untuk ahli jaringan komputer guna melihat kesamaan performa dan efektivitasnya.
	\end{enumerate}
\end{frame}


% ------------------------------
\begin{frame}{Mengapa Abstraksi Tugas Diperlukan?}
	\begin{enumerate}
		\item Membantu Pemilihan Teknik yang Tepat \\
		Membantu desainer memetakan kebutuhan pengguna ke Visual Channel yang paling sesuai. 
		\begin{itemize}
			\item Jika tugas diabstraksikan sebagai Identify Distribution, maka histogram mungkin sangat efektif.
			\item Jika tugas diabstraksikan sebagai Identify Correlation, maka scatterplot lebih unggul.
			\item Tanpa abstraksi, mungkin hanya membuat grafik berdasarkan "apa yang biasa dipakai di bidang itu", bukan berdasarkan apa yang secara kognitif paling efektif bagi mata manusia.
		\end{itemize}
		\item Menentukan Ruang Desain (Design Space) \\
		Abstraksi membatasi pilihan. Mengetahui bahwa tugas pengguna adalah Compare Trends, kita bisa langsung mengeliminasi banyak jenis grafik yang tidak mendukung perbandingan tren, sehingga proses desain menjadi lebih fokus dan efisien.
	\end{enumerate}
\end{frame}

\begin{frame}{Domain Task vs. Abstract Task}
    \begin{block}{Definisi}
        \textbf{Domain Task:} Kebutuhan pengguna dalam bahasa sehari-hari/bidang spesifik. \\
        \textbf{Abstract Task:} Terjemahan tugas ke dalam kosakata kerangka kerja Munzner.
    \end{block}

    \vspace{0.3cm}
    \centering
    \small
    \begin{tabular}{lp{4cm}p{4cm}}
        \toprule
        \textbf{Domain} & \textbf{Domain Task} & \textbf{Abstract Task} \\ \midrule
        Kesehatan & "Cari pasien dengan suhu tubuh aneh." & \textbf{Locate Outliers} \\ \addlinespace
        Ekonomi & "Lihat pola harga emas tahunan." & \textbf{Identify Trends} \\ \addlinespace
        Sosial & "Siapa teman dari si Budi?" & \textbf{Explore Topology} \\ \addlinespace
        Retail & "Bandingkan penjualan Cabang A dan B." & \textbf{Compare Attributes} \\ \bottomrule
    \end{tabular}
    
    \vspace{0.4cm}
    \textit{Kesimpulan: Abstraksi membantu kita menemukan alat visual yang tepat tanpa peduli apa bidangnya.}
\end{frame}

\begin{frame}{Who: Designer or User?}
\begin{itemize}
	\item Peran Desainer vs. Pengguna
	\begin{itemize}
		\item Pengguna (User): Fokus pada masalah domain mereka. Mereka bicara tentang "gen", "profit", "pasien", atau "keamanan jaringan". Mereka biasanya menjelaskan tugas dalam kalimat yang sangat spesifik dan terkadang tidak terstruktur.
		\item Desainer (Designer): Fokus pada penerjemahan. Tugas desainer bukan sekadar "menggambar grafik", tetapi menganalisis niat pengguna dan mengubahnya menjadi Abstract Task (seperti Compare, Summarize, atau Locate)
	\end{itemize}
	\item Tantangan: "The Gap" (Celah Pemahaman)
	\begin{itemize}
		\item Pengguna mungkin tidak tahu apa yang mungkin dilakukan. Mereka meminta tabel karena mereka tidak tahu bahwa Scatterplot bisa menunjukkan korelasi lebih cepat.
		\item Desainer mungkin tidak paham konteks domain. Tanpa abstraksi, desainer mungkin membuat grafik yang secara estetika bagus, tapi tidak menjawab pertanyaan inti sang dokter atau analis keuangan.
	\end{itemize}
\end{itemize}
\end{frame}

    \vspace{0.3cm}
    \textit{Designer membantu User melihat apa yang sebelumnya tidak terlihat.}
\end{frame}

\begin{frame}{Actions: Hierarki Tindakan Pengguna}
    \begin{block}{Definisi}
        \textit{Actions} mendefinisikan \textbf{bagaimana} pengguna berinteraksi dengan data, mulai dari niat umum hingga pertanyaan spesifik.
    \end{block}

    \vspace{0.3cm}
    \begin{columns}[t]
        \begin{column}{0.31\textwidth}
            \textbf{1. Analyze} \\
            \scriptsize (Niat Umum)
            \begin{itemize}
                \item \textbf{Consume:} Discover, Present, Enjoy.
                \item \textbf{Produce:} Annotate, Record, Derive.
            \end{itemize}
        \end{column}
        \begin{column}{0.31\textwidth}
            \textbf{2. Search} \\
            \scriptsize (Cara Menemukan)
            \begin{itemize}
                \item \textbf{Lookup:} Tahu target \& lokasi.
                \item \textbf{Locate:} Tahu target, cari lokasi.
                \item \textbf{Browse/Explore.}
            \end{itemize}
        \end{column}
        \begin{column}{0.31\textwidth}
            \textbf{3. Query} \\
            \scriptsize (Kedalaman Pertanyaan)
            \begin{itemize}
                \item \textbf{Identify:} Satu item.
                \item \textbf{Compare:} Antar item.
                \item \textbf{Summarize:} Semua data.
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

% ------------------------------

\begin{frame}{Actions: Hierarki Tindakan Pengguna - Analyze}
Niat paling mendasar pengguna. Dibagi menjadi dua jalur
\begin{itemize}
	\item Consume (Mengonsumsi): Menggunakan visualisasi yang sudah ada.
	\begin{itemize}
		\item Discover: Mencari wawasan baru yang belum diketahui sebelumnya (Eksploratif).
		\item Present: Menyampaikan temuan yang sudah diketahui kepada orang lain (Komunikasi/Storytelling).
		\item Enjoy: Mengamati data untuk kesenangan atau ketertarikan personal (Casual Viz).
	\end{itemize}
	\item Produce (Memproduksi): Pengguna menghasilkan informasi baru saat berinteraksi dengan visualisasi.
	\begin{itemize}
		\item Annotate: Menambahkan catatan atau label pada elemen visual.
		\item Record: Menyimpan riwayat jejak analisis atau membuat screenshot temuan untuk digunakan nanti.
		\item Derive: Membuat atribut baru dari data yang ada (misal: menghitung selisih harga dari kolom harga asli).
	\end{itemize}
\end{itemize}
\end{frame}
% ------------------------------

\begin{frame}{Actions: Hierarki Tindakan Pengguna - Search}
Bagaimana cara pengguna menemukan target? Munzner membedakannya berdasarkan apakah pengguna tahu apa yang dicari (Target) dan di mana letaknya (Location).
\vspace{-0.2cm}
\begin{table}
    \begin{tabular}{@{} lcc p{6cm} @{}}
        \toprule
        \textbf{Tipe Search} & \textbf{Target?} & \textbf{Lokasi ?} & \textbf{Contoh Analogi} \\ 
        \midrule
        \textbf{Lookup} & Ya & Ya & Mencari harga saham Apple di tanggal hari ini. \\ \addlinespace
        \textbf{Locate} & Ya & Tidak & Mencari di mana letak koordinat kota yang suhunya $> 40^\circ$C. \\ \addlinespace
        \textbf{Browse} & Tidak & Ya & Melihat-lihat tren penjualan di wilayah Jawa Barat saja. \\ \addlinespace
        \textbf{Explore} & Tidak & Tidak & Menjelajah seluruh data untuk melihat apakah ada yang menarik. \\ 
        \bottomrule
    \end{tabular}
\end{table}

\end{frame}

\begin{frame}{Actions: Hierarki Tindakan Pengguna - Query (Bertanya)}
Apa yang dilakukan terhadap item tersebut?
\begin{itemize}
	\item Identify: Hanya melihat karakteristik dari satu item tunggal.
	\item Compare: Melihat perbedaan atau persamaan antara dua atau lebih item.
	\item Summarize: Melihat gambaran besar dari seluruh data yang tersedia.
\end{itemize}
\end{frame}

\begin{frame}{Implementasi Actions: Heatmap Penjualan}
    \textbf{Dataset:} Transaksi per jam selama satu minggu.
    
    \begin{itemize}
        \item \textbf{Search - Locate:} 
        Mata pengguna mencari sel dengan warna paling merah (paling laku) untuk mengetahui kapan jam puncak terjadi.
        
        \item \textbf{Query - Compare:} 
        Membandingkan intensitas warna antara baris "Senin" dan baris "Minggu" untuk melihat perbedaan perilaku belanja.
        
        \item \textbf{Analyze - Derive:} 
        Mengubah data mentah transaksi menjadi data "Rata-rata Kunjungan per Jam" sebelum ditampilkan di heatmap.
    \end{itemize}
    
    \vspace{0.3cm}
    \begin{quote}
        "Visualisasi yang efektif memungkinkan pengguna melakukan \textit{Locate} dan \textit{Compare} dengan beban kognitif yang rendah."
    \end{quote}
\end{frame}


% ------------------------------

\begin{frame}{Implementasi Actions: Analyze }
    \textbf{Dataset:} dataset penjualan selama 1 tahun.
    \begin{itemize}
    	\item Consume $\rightarrow$ Discover: Seorang analis melihat line chart penjualan bulanan dan baru menyadari bahwa ada lonjakan drastis setiap bulan Oktober (menemukan wawasan baru: ternyata ada tren musiman).
    	\item Consume $\rightarrow$ Present: Manajer menggunakan grafik yang sama untuk membuktikan kepada direksi bahwa strategi diskon mereka berhasil menaikkan profit.
    	\item Produce $\rightarrow$ Derive: Dari data "Harga Jual" dan "Harga Modal", analis membuat atribut baru yaitu "Profit" ($Profit = Jual - Modal$) untuk kemudian divisualisasikan.
    \end{itemize}
    
\end{frame}

% ------------------------------

\begin{frame}{Implementasi Actions: Search }
Visualisasi Heatmap Penjualan (Sumbu X: Jam, Sumbu Y: Hari).
\begin{itemize}
	\item Lookup. "Berapa penjualan pada hari Sabtu jam 8 malam?" (Target tahu: Sabtu jam 8, Lokasi tahu: baris Sabtu, kolom jam 20.00).
	\item Locate. "Jam berapa yang penjualannya paling sepi (warna paling terang)?" (Target tahu: nilai terendah, Lokasi tidak tahu: harus mencari di seluruh kotak heatmap).
	\item Browse. "Bagaimana pola penjualan khusus di akhir pekan (Sabtu-Minggu)?" (Lokasi tahu: dua baris terbawah, Target tidak pasti: hanya ingin melihat-lihat pola di area tersebut).
	\item Explore. "Apakah ada sesuatu yang aneh atau menarik dari seluruh heatmap ini?" (Target dan lokasi tidak tahu: memindai seluruh visualisasi tanpa tujuan spesifik).
\end{itemize}
\end{frame}


% ------------------------------

\begin{frame}{Implementasi Actions: Query }
\begin{itemize}
	\item Identify: "Oh, ternyata di titik ini (Senin jam 10 pagi) penjualannya hanya Rp 500.000." (Fokus pada satu sel).
	\item Compare: "Mana yang lebih ramai, Senin pagi atau Jumat pagi?" (Membandingkan dua sel atau lebih).
	\item Summarize: "Secara keseluruhan, toko kita paling ramai di hari kerja pada jam makan siang." (Menyimpulkan seluruh area heatmap).
\end{itemize}

\end{frame}

\begin{frame}{Targets: Apa yang Dicari?}
    \begin{block}{Definisi}
        \textit{Targets} adalah aspek spesifik dari data yang ingin diperiksa/ditemukan/ diamati oleh pengguna melalui visualisasi.
    \end{block}

    \vspace{0.3cm}
    \begin{columns}[t]
        \begin{column}{0.48\textwidth}
            \textbf{High-Level Targets - Target Seluruh Data:}
            \begin{itemize}
                \item \textbf{Trends:} Pola kenaikan/penurunan.
                \item \textbf{Outliers:} Data yang menyimpang.
                \item \textbf{Features:} Struktur unik data.
            \end{itemize}
        \end{column}
        
        \begin{column}{0.48\textwidth}
            \textbf{Target Atribut:}
            \begin{itemize}
                \item \textbf{Distribution:} Sebaran nilai.
                \item \textbf{Extremes:} Nilai Max/Min.
                \item \textbf{Correlation:} Hubungan antar variabel.
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Targets: Apa yang Dicari?}
\textbf{High-Level Targets - Target Seluruh Data}
\begin{itemize}
	\item Trends: Pola umum atau kecenderungan data (misal: naik, turun, stabil, atau siklik).\\
	Contoh: "Penjualan iPhone selalu meningkat di kuartal keempat setiap tahun."
	\item Outliers: Titik data yang menyimpang secara signifikan dari pola umum.\\
	Contoh: "Ada satu transaksi senilai 1 Miliar di toko kecil, sementara rata-rata transaksi lainnya hanya 100 ribu."
	\item Features: Karakteristik atau bentuk khusus yang menarik (misal: ada dua gundukan pada distribusi data).\\
	Contoh: "Data menunjukkan adanya dua kelompok pembeli yang berbeda (bimodal)."
\end{itemize}
\end{frame}


% ------------------------------
\begin{frame}{Targets: Apa yang Dicari?}
\textbf{Target Atribut}
\begin{itemize}
	\item One Attribute:\\
	Distribution: Bagaimana nilai-nilai tersebar (apakah berkumpul di tengah atau menyebar merata?).\\
	Extremes: Mencari nilai absolut tertinggi (maksimum) atau terendah (minimum).
	\item Many Attributes: \\
	Correlation: Apakah dua atribut saling memengaruhi? (Jika X naik, apakah Y juga naik?). \\
	Similarity: Mencari item-item yang memiliki kemiripan karakteristik. \\
\end{itemize}
\end{frame}

% ------------------------------
\begin{frame}{Targets: Apa yang Dicari?}
\textbf{Targets Terkait Data Relasional (Network/Tree)}. Jika dataset kita berupa jaringan (siapa berteman dengan siapa).
\begin{itemize}
	\item Topology: Memahami struktur hubungan (siapa pusatnya, siapa yang terisolasi, atau bagaimana kelompok-kelompok terbentuk).
	\item Paths: Mencari jalur atau koneksi antara dua titik dalam jaringan.
\end{itemize}

\textbf{Targets Terkait Data Spasial (Spatial Data)}.Jika data memiliki bentuk fisik atau lokasi.
\begin{itemize}
	\item Shape: Memahami bentuk geografis atau struktur 3D (misal: bentuk tumor pada data medis atau bentuk pegunungan pada peta).
\end{itemize}
\end{frame}
